  <script>
    // ==========================================
    // 1. 全域變數與狀態管理
    // ==========================================
    const apiKey = "";
    const appId = typeof __app_id !== 'undefined' ? __app_id : 'menu-parser-spa';
    
    // 狀態管理物件 (預設值已更新為「雯雯小吃」與「吃益生菌提升免疫力」)
    const state = {
      base64Image: null,
      mimeType: null,
      menuTitle: "雯雯小吃",
      menuSubtitle: "吃益生菌提升免疫力",
      parsedItems: [], // 僅存放 AI 解析出的品項陣列
      activeTab: 'json' // 'json' | 'table'
    };

    // ==========================================
    // 2. DOM 元素選取
    // ==========================================
    const dropZone = document.getElementById('drop-zone');
    const fileInput = document.getElementById('file-input');
    const uploadPrompt = document.getElementById('upload-prompt');
    const previewContainer = document.getElementById('preview-container');
    const imagePreview = document.getElementById('image-preview');
    const clearBtn = document.getElementById('clear-btn');
    const recognizeBtn = document.getElementById('recognize-btn');
    const scanLine = document.getElementById('scan-line');

    const inputTitle = document.getElementById('input-title');
    const inputSubtitle = document.getElementById('input-subtitle');

    const tabJson = document.getElementById('tab-json');
    const tabTable = document.getElementById('tab-table');
    const contentJson = document.getElementById('content-json');
    const contentTable = document.getElementById('content-table');
    const outputPlaceholder = document.getElementById('output-placeholder');

    const jsonCode = document.getElementById('json-code');
    const tableBody = document.getElementById('table-body');
    const itemsCountIndicator = document.getElementById('items-count-indicator');
    
    const copyBtn = document.getElementById('copy-btn');
    const downloadBtn = document.getElementById('download-btn');
    const toastContainer = document.getElementById('toast-container');

    // ==========================================
    // 3. 通用輔助函式：自訂 Toast 提示
    // ==========================================
    function showToast(message, type = 'success') {
      const toast = document.createElement('div');
      toast.className = `flex items-center gap-3 px-4 py-3 rounded-xl border shadow-xl backdrop-blur-md transform translate-y-2 opacity-0 transition-all duration-300 pointer-events-auto text-xs ${
        type === 'success' 
          ? 'bg-slate-900/90 border-emerald-500/30 text-emerald-300' 
          : type === 'error'
          ? 'bg-slate-900/90 border-rose-500/30 text-rose-300'
          : 'bg-slate-900/90 border-indigo-500/30 text-indigo-300'
      }`;

      const icon = type === 'success' ? 'check-circle' : type === 'error' ? 'alert-triangle' : 'info';
      toast.innerHTML = `
        <i data-lucide="${icon}" class="w-4.5 h-4.5 flex-shrink-0"></i>
        <span class="font-medium">${message}</span>
      `;
      
      toastContainer.appendChild(toast);
      lucide.createIcons();

      // 進場動畫
      setTimeout(() => {
        toast.classList.remove('translate-y-2', 'opacity-0');
      }, 10);

      // 退場自毀
      setTimeout(() => {
        toast.classList.add('translate-y-2', 'opacity-0');
        setTimeout(() => {
          toast.remove();
        }, 300);
      }, 3500);
    }

    // ==========================================
    // 4. 影像處理與上傳事件
    // ==========================================
    dropZone.addEventListener('click', () => fileInput.click());

    ['dragenter', 'dragover'].forEach(eventName => {
      dropZone.addEventListener(eventName, (e) => {
        e.preventDefault();
        dropZone.classList.add('border-indigo-500', 'bg-indigo-950/20');
      }, false);
    });

    ['dragleave', 'drop'].forEach(eventName => {
      dropZone.addEventListener(eventName, (e) => {
        e.preventDefault();
        dropZone.classList.remove('border-indigo-500', 'bg-indigo-950/20');
      }, false);
    });

    dropZone.addEventListener('drop', (e) => {
      const dt = e.dataTransfer;
      const files = dt.files;
      if (files.length > 0) {
        handleFile(files[0]);
      }
    });

    fileInput.addEventListener('change', (e) => {
      if (e.target.files.length > 0) {
        handleFile(e.target.files[0]);
      }
    });

    clearBtn.addEventListener('click', (e) => {
      e.stopPropagation();
      resetUpload();
    });

    function handleFile(file) {
      if (!file.type.startsWith('image/')) {
        showToast('請匯入正確的影像格式檔！', 'error');
        return;
      }

      state.mimeType = file.type;
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onloadend = () => {
        const base64Data = reader.result.split(',')[1];
        state.base64Image = base64Data;
        
        imagePreview.src = reader.result;
        uploadPrompt.classList.add('hidden');
        previewContainer.classList.remove('hidden');
        clearBtn.classList.remove('hidden');
        recognizeBtn.removeAttribute('disabled');
        
        showToast('影像載入完成，可以開始解析。', 'info');
      };
    }

    function resetUpload() {
      state.base64Image = null;
      state.mimeType = null;
      fileInput.value = '';
      imagePreview.src = '';
      uploadPrompt.classList.remove('hidden');
      previewContainer.classList.add('hidden');
      clearBtn.classList.add('hidden');
      recognizeBtn.setAttribute('disabled', 'true');
    }

    // ==========================================
    // 5. 即時雙向連動渲染機制 (雙向綁定)
    // ==========================================
    inputTitle.addEventListener('input', (e) => {
      state.menuTitle = e.target.value;
      renderOutput();
    });

    inputSubtitle.addEventListener('input', (e) => {
      state.menuSubtitle = e.target.value;
      renderOutput();
    });

    function renderOutput() {
      // 組合最終呈現的 JSON 結構
      const finalJSON = {
        title: state.menuTitle || "",
        subtitle: state.menuSubtitle || "",
        items: state.parsedItems
      };

      const jsonString = JSON.stringify(finalJSON, null, 2);
      jsonCode.textContent = jsonString;

      renderTable(finalJSON.items);
      itemsCountIndicator.textContent = `品項總數：${finalJSON.items.length} 項`;

      if (state.parsedItems.length > 0) {
        copyBtn.removeAttribute('disabled');
        downloadBtn.removeAttribute('disabled');
        outputPlaceholder.classList.add('hidden');
        
        if (state.activeTab === 'json') {
          contentJson.classList.remove('hidden');
          contentTable.classList.add('hidden');
        } else {
          contentJson.classList.add('hidden');
          contentTable.classList.remove('hidden');
        }
      } else {
        copyBtn.setAttribute('disabled', 'true');
        downloadBtn.setAttribute('disabled', 'true');
        outputPlaceholder.classList.remove('hidden');
        contentJson.classList.add('hidden');
        contentTable.classList.add('hidden');
      }
    }

    function renderTable(items) {
      tableBody.innerHTML = '';
      if (!items || items.length === 0) {
        tableBody.innerHTML = `
          <tr>
            <td colspan="4" class="py-6 text-center text-slate-500">暫無品項資料</td>
          </tr>
        `;
        return;
      }

      items.forEach((item, index) => {
        const tr = document.createElement('tr');
        tr.className = "hover:bg-slate-900/40 transition-colors";
        tr.innerHTML = `
          <td class="py-3 px-4 text-slate-500 font-mono">${index + 1}</td>
          <td class="py-3 px-4 font-medium text-slate-200">${escapeHtml(item.name)}</td>
          <td class="py-3 px-4 text-right text-emerald-400 font-mono font-semibold">${item.price !== undefined ? item.price : '-'}</td>
          <td class="py-3 px-4 text-center text-slate-400">${escapeHtml(item.unit) || '<span class="text-slate-600">-</span>'}</td>
        `;
        tableBody.appendChild(tr);
      });
    }

    // 防禦 XSS
    function escapeHtml(str) {
      if (!str) return '';
      return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;").replace(/'/g, "&#039;");
    }

    // ==========================================
    // 6. 分頁切換 (Tabs)
    // ==========================================
    tabJson.addEventListener('click', () => switchTab('json'));
    tabTable.addEventListener('click', () => switchTab('table'));

    function switchTab(tab) {
      state.activeTab = tab;
      if (tab === 'json') {
        tabJson.className = "flex items-center gap-2 px-4 py-1.5 text-xs font-semibold rounded-md transition-all text-white bg-indigo-600/90 shadow-sm shadow-indigo-500/10";
        tabTable.className = "flex items-center gap-2 px-4 py-1.5 text-xs font-semibold rounded-md transition-all text-slate-400 hover:text-slate-200";
        
        if (state.parsedItems.length > 0) {
          contentJson.classList.remove('hidden');
          contentTable.classList.add('hidden');
        }
      } else {
        tabTable.className = "flex items-center gap-2 px-4 py-1.5 text-xs font-semibold rounded-md transition-all text-white bg-indigo-600/90 shadow-sm shadow-indigo-500/10";
        tabJson.className = "flex items-center gap-2 px-4 py-1.5 text-xs font-semibold rounded-md transition-all text-slate-400 hover:text-slate-200";
        
        if (state.parsedItems.length > 0) {
          contentJson.classList.add('hidden');
          contentTable.classList.remove('hidden');
        }
      }
    }

    // ==========================================
    // 7. 指數型退避重試請求與 AI 串接
    // ==========================================
    async function fetchWithRetry(url, options, maxRetries = 5) {
      let delay = 1000;
      for (let i = 0; i < maxRetries; i++) {
        try {
          const response = await fetch(url, options);
          if (response.ok) {
            return await response.json();
          }
          throw new Error(`伺服器回應錯誤狀態碼: ${response.status}`);
        } catch (error) {
          if (i === maxRetries - 1) {
            throw error;
          }
          await new Promise(resolve => setTimeout(resolve, delay));
          delay *= 2;
        }
      }
    }

    recognizeBtn.addEventListener('click', async () => {
      if (!state.base64Image) {
        showToast('請先上傳圖片！', 'error');
        return;
      }

      setLoadingState(true);
      showToast('開始傳送 AI 多模態辨識請求，請稍候...', 'info');

      // 構建 Gemini 2.5 Flash API Payload
      const apiEndpoint = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
      
      const payload = {
        contents: [
          {
            parts: [
              { 
                text: "請分析辨識這張菜單。請辨識店家名稱（如餐廳招牌名稱）設為 title，店家標語或副標（如菜單上的第二行宣傳文字、 slogan 或子標題）設為 subtitle。如果無法在菜單圖片中找到店家名稱或標語，請回傳空字串 \"\"。另外，請將菜單上的所有品項詳細轉換為結構化資料，存入 'items' 陣列。若品項有大小份（如大、小、中）、或是口味規格（如：原味、辣味），請務必拆分為獨立的物件。price 必須為整數數字，拒絕區間字串。unit 則是該品項的單位（如：碗、杯、份，若無則填空字串）。" 
              },
              {
                inlineData: {
                  mimeType: state.mimeType,
                  data: state.base64Image
                }
              }
            ]
          }
        ],
        generationConfig: {
          responseMimeType: "application/json",
          responseSchema: {
            type: "OBJECT",
            properties: {
              title: { 
                type: "STRING", 
                description: "圖片中辨識出的店家名稱/餐廳招牌 (若找不到請回傳空字串)" 
              },
              subtitle: { 
                type: "STRING", 
                description: "圖片中辨識出的店名標語、Slogan 或副標題 (若找不到請回傳空字串)" 
              },
              items: {
                type: "ARRAY",
                items: {
                  type: "OBJECT",
                  properties: {
                    name: { 
                      type: "STRING", 
                      description: "品項名稱 (若有區分大小份、口味，請拆分為獨立物件)" 
                    },
                    price: { 
                      type: "INTEGER", 
                      description: "必須為整數數字，拒絕區間字串" 
                    },
                    unit: { 
                      type: "STRING", 
                      description: "單位 (如：碗、杯、份，若無則填空字串)" 
                    }
                  },
                  required: ["name", "price", "unit"]
                }
              }
            },
            required: ["title", "subtitle", "items"]
          }
        }
      };

      try {
        const responseData = await fetchWithRetry(apiEndpoint, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          },
          body: JSON.stringify(payload)
        });

        const textResponse = responseData.candidates?.[0]?.content?.parts?.[0]?.text;
        if (!textResponse) {
          throw new Error('API 回傳格式異常：未能成功提取文本內容');
        }

        const parsedResult = JSON.parse(textResponse);

        // ==========================================
        // 8. 智慧標題與副標題優先順序與 Fallback 機制 (強化連動)
        // ==========================================
        
        // 處理 頁面標題 (Title)
        if (parsedResult.title && parsedResult.title.trim() !== "") {
          state.menuTitle = parsedResult.title.trim();
          inputTitle.value = state.menuTitle; // 同步更新輸入框
          inputTitle.dispatchEvent(new Event('input')); // 強制觸發雙向連動更新事件
          showToast(`已偵測並套用店名：${state.menuTitle}`, 'success');
        } else {
          // 找不到則採用使用者預設值/原輸入
          state.menuTitle = inputTitle.value || "雯雯小吃";
          inputTitle.dispatchEvent(new Event('input')); // 確保派發事件更新
          showToast('未偵測到店名，採用預設標題', 'info');
        }

        // 處理 副標題 (Subtitle)
        if (parsedResult.subtitle && parsedResult.subtitle.trim() !== "") {
          state.menuSubtitle = parsedResult.subtitle.trim();
          inputSubtitle.value = state.menuSubtitle; // 同步更新輸入框
          inputSubtitle.dispatchEvent(new Event('input')); // 強制觸發雙向連動更新事件
          showToast(`已偵測並套用標語：${state.menuSubtitle}`, 'success');
        } else {
          // 找不到則採用使用者預設值/原輸入
          state.menuSubtitle = inputSubtitle.value || "";
          inputSubtitle.dispatchEvent(new Event('input')); // 確保派發事件更新
          if (state.menuSubtitle) {
            showToast('未偵測到標語，採用預設副標題', 'info');
          }
        }

        // 保存解析出的品項列表
        state.parsedItems = parsedResult.items || [];
        
        // 渲染結果
        renderOutput();
        showToast(`辨識成功！共解析出 ${state.parsedItems.length} 項品項。`, 'success');

      } catch (err) {
        console.error(err);
        showToast(`辨識失敗: ${err.message}`, 'error');
      } finally {
        setLoadingState(false);
      }
    });

    downloadBtn.addEventListener('click', () => {
      const jsonText = jsonCode.textContent;
      if (!jsonText) return;

      const blob = new Blob([jsonText], { type: 'application/json;charset=utf-8;' });
      
      const now = new Date();
      const pad = (num) => String(num).padStart(2, '0');
      const timestampStr = `${now.getFullYear()}${pad(now.getMonth() + 1)}${pad(now.getDate())}_${pad(now.getHours())}${pad(now.getMinutes())}`;
      
      const fileName = `menu_structured_${timestampStr}.json`;

      const link = document.createElement('a');
      const url = URL.createObjectURL(blob);
      link.setAttribute('href', url);
      link.setAttribute('download', fileName);
      link.style.visibility = 'hidden';
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      
      showToast(`已成功下載：${fileName}`, 'success');
    });
